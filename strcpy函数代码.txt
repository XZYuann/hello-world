#include<stdio.h>
#include <malloc.h>
char* StrCpy(char* src);
int main() {
}
char* StrCpy(char* src)
	{
		//1、计算src的长度 len
	int len = sizeof(src) / sizeof(src[0]);
		//2、用malloc申请长度为 len+1的内存空间
	char* size = (char*)malloc((len+1)*sizeof(char));
		//3、将src字符串逐个字符复制到新申请的内存空间
		//4、新申请内存空间的最后一位放入'\0'
	for (int i=0;i<len+1;i++)
	{
		if (i<len)
		{
		size[i] = src[i];
		}
		else
		{
			size[i] = '\0';
		}
	}
		//5、返回新申请的内存空间指针
	return size;
	}
